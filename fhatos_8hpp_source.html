247 static void MAIN_LOG(const LOG_TYPE type, const char *format, ...) {
248 if ((uint8_t) type < (uint8_t) LOG_TYPE::FOS_LOGGING)
249 return;
250 va_list arg;
251 va_start(arg, format);
252 char temp[128];
253 char *buffer = temp;
254 const size_t len = vsnprintf(temp, sizeof(temp), format, arg);
255 va_end(arg);
256 if (len > sizeof(temp) - 1) {
257 buffer = new (std::nothrow) char[len + 1];
258 if (!buffer) {
259 return;
260 }
261 va_start(arg, format);
262 vsnprintf(buffer, len + 1, format, arg);
263 va_end(arg);
264 }
265 if (type == NONE)
266 FOS_DEFAULT_PRINTER::singleton()->print("");
267 else if (type == ERROR)
268 FOS_DEFAULT_PRINTER::singleton()->print("!r[ERROR]!! ");
269 else if (type == INFO)
270 FOS_DEFAULT_PRINTER::singleton()->print("!g[INFO]!! ");
271 else
272 FOS_DEFAULT_PRINTER::singleton()->print("!y[DEBUG]!! ");
273 FOS_DEFAULT_PRINTER::singleton()->print(buffer);
274 if (buffer != temp) {
275 delete[] buffer;
276 }
277 }
