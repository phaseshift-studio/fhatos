558 class Bytecode final : public Obj, public IDed {
559 protected:
560 const List<Inst *> *_value;
561 Map<const fURI, const Obj *> TYPE_CACHE;

563 public:
564 explicit Bytecode(const List<Inst *> *list, const ID &id = ID(*UUID::singleton()->mint(7))) :
565 Obj(OType::BYTECODE), IDed(id), _value(list) {
566 for (auto *inst: *this->_value) {
567 inst->bcode(this);

569 }
570
571 explicit Bytecode(const ID &id = ID(*UUID::singleton()->mint(7))) : Bytecode(new List<Inst *>, id) {}
572
573 const Inst *nextInst(const Inst *currentInst) const {
574 if (currentInst->isNoInst())
575 return currentInst;
576 bool found = false;
577 for (const Inst *inst: *this->value()) {
578 if (found)
579 return inst;
580 if (inst == currentInst)
581 found = true;
582 }
583 return NoInst::singleton();
584 }
585
586 const Obj *apply(const Obj *obj) const override {
587 auto currentObj = const_cast<Obj *>(obj);
588 if (currentObj->isNoObj())
589 return currentObj;
590 for (const Inst *inst: *this->_value) {
591 currentObj = (Obj *) inst->apply(currentObj);
592 if (currentObj->isNoObj())
593 break;
594 }
595 return currentObj;
596 }

598 const List<Inst *> *value() const { return this->_value; }

600 ptr<Bytecode> addInst(Inst *inst) const {
601 auto *list = new List<Inst *>();
602 const ptr<Bytecode> bcode = share(Bytecode(list, this->id()));
603 for (const auto prevInst: *this->_value) {
604 list->push_back(prevInst->bcode(bcode.get()));
605 }
606 list->push_back(inst->bcode(bcode.get()));
607 return bcode;
608 }
