58 const RESPONSE_CODE publish(const Message &message) override {
59 return MUTEX_SUBSCRIPTIONS.read<RESPONSE_CODE>([this, message] {
61 RESPONSE_CODE _rc = message.retain ? OK : NO_TARGETS;
62 const ptr<Message> mess_ptr = share<Message>(Message(message));
63 for (const auto &subscription: SUBSCRIPTIONS) {
64 if (subscription->pattern.matches(message.target)) {
65 try {
66 if (subscription->mailbox) {
67 _rc = subscription->mailbox->push(share<Mail>(Mail(subscription, mess_ptr))) ? OK : ROUTER_ERROR;
68 LOG(DEBUG, "Message from %s delivered to %s\n", message.source.toString().c_str(), subscription->source.toString().c_str());
69 if (subscription->mailbox->size() > FOS_MAILBOX_WARNING_SIZE) {
70 LOG(ERROR, "Actor mailbox size is beyond warning size of %i: [size:%i]\n", FOS_MAILBOX_WARNING_SIZE,
71 subscription->mailbox->size());
72 }
73 } else {
74 subscription->onRecv(mess_ptr);
75 _rc = OK;
76 }
77
78 } catch (const fError &e) {
79 LOG_EXCEPTION(e);
80 _rc = MUTEX_TIMEOUT;
81 }
82 }
83 }
84 if (message.retain) {
85 MUTEX_RETAIN.lockUnlock<void *>([this, mess_ptr]() {
86 if (RETAINS.count(mess_ptr->target))
87 RETAINS.erase(mess_ptr->target);
88 RETAINS.insert({mess_ptr->target, mess_ptr});
89 return nullptr;
90 });
91 LOG(DEBUG, "Total number of retained messages [size:%i]\n", RETAINS.size());
92 }
93 LOG_PUBLISH(_rc, message);
94 return _rc;
95 });
96 }

98 const RESPONSE_CODE subscribe(const Subscription &subscription) override {
99 try {
100 return *MUTEX_SUBSCRIPTIONS.write<RESPONSE_CODE>([this, subscription]() {
101 RESPONSE_CODE _rc = OK;
102 for (const auto &sub: SUBSCRIPTIONS) {
103 if (sub->source.equals(subscription.source) && sub->pattern.equals(subscription.pattern)) {
104 _rc = REPEAT_SUBSCRIPTION;
105 break;
106 }
107 }
108 if (!_rc) {
109 const ptr<Subscription> sub_ptr = share<Subscription>(subscription);
110 SUBSCRIPTIONS.push_back(sub_ptr);
111 LOG_SUBSCRIBE(_rc, sub_ptr);
112 MUTEX_RETAIN.lockUnlock<void *>([this, subscription, sub_ptr]() {
113 LOG(DEBUG, "Processing retain messages [size:%i]\n", RETAINS.size());
114 for (const auto &[target, message]: RETAINS) {
115 if (target.matches(subscription.pattern)) {
116 if (subscription.mailbox) {
117 subscription.mailbox->push(share<Mail>(Mail(sub_ptr, message)));
118 } else {
119 subscription.onRecv(message);
120 }
121 }
122 }
123 return nullptr;
124 });
125 }
126 return share<RESPONSE_CODE>(_rc);
127 });
128 } catch (const fError &e) {
129 LOG_EXCEPTION(e);
130 return MUTEX_TIMEOUT;
131 }
132 }

141 ptr<RESPONSE_CODE> unsubscribeX(const ID &source, const Pattern *pattern) {
142 try {
143 return MUTEX_SUBSCRIPTIONS.write<RESPONSE_CODE>([this, source, pattern]() {
144 const uint16_t size = SUBSCRIPTIONS.size();
145 SUBSCRIPTIONS.erase(remove_if(SUBSCRIPTIONS.begin(), SUBSCRIPTIONS.end(),
146 [source, pattern](const auto &sub) {
147 return sub->source.equals(source) &&
148 (nullptr == pattern || sub->pattern.equals(*pattern));
149 }),
150 SUBSCRIPTIONS.end());
151 const auto _rc2 =
152 share<RESPONSE_CODE>(((SUBSCRIPTIONS.size() < size) || pattern == nullptr) ? OK : NO_SUBSCRIPTION);
153 LOG_UNSUBSCRIBE(*_rc2, source, pattern);
154 return _rc2;
155 });
156 } catch (const fError &e) {
157 LOG_EXCEPTION(e);
158 return share<RESPONSE_CODE>(MUTEX_TIMEOUT);
159 }
160 }
