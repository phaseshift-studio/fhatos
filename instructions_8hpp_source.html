67 template<typename ROUTER = FOS_DEFAULT_ROUTER>
68 class AsInst final : public OneToOneInst {
69 public:
70 explicit AsInst(const OBJ_OR_BYTECODE &utype) :
71 OneToOneInst({"as", {utype}, [this](const Obj *obj) {
72 const UType utype = this->arg(0)->apply(obj)->template as<Uri>()->value();
73 const Obj *typeDefinition = this->_bcode->template getType<ROUTER>(utype);
74 if (typeDefinition->type() != OType::BYTECODE && typeDefinition->type() != obj->type())
75 return NoObj::singleton()->obj();
76 if (typeDefinition->apply(obj)->isNoObj()) {
77 LOG(ERROR, "%s is not a !y%s!!%s\n", obj->toString().c_str(), utype.toString().c_str(),
78 typeDefinition->toString().c_str());
79 return NoObj::singleton()->obj();
80 }
81 return cast(obj, utype);
82 }}) {}
83 };
