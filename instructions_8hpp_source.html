201 static const Inst_p to_inst(const fURI &type, const List<Obj_p> &args) {
202 if (type == INST_FURI->resolve("start") || type == INST_FURI->resolve("__"))
203 return Insts::start(args);
204 if (type == INST_FURI->resolve("map"))
205 return Insts::map(args.at(0));
206 if (type == INST_FURI->resolve("filter"))
207 return Insts::filter(args.at(0));
208 if (type == INST_FURI->resolve("side"))
209 return Insts::side(args.at(0));
210 if (type == INST_FURI->resolve("noop"))
211 return Insts::noop();
212 if (type == INST_FURI->resolve("as"))
213 return Insts::as(args.at(0));
214 if (type == INST_FURI->resolve("define"))
215 return Insts::define(args.at(0), args.at(1));
216 if (type == INST_FURI->resolve("type"))
217 return Insts::type();
218 if (type == INST_FURI->resolve("is"))
219 return Insts::is(args.at(0));
220 if (type == INST_FURI->resolve("plus"))
221 return Insts::plus(args.at(0));
222 if (type == INST_FURI->resolve("mult"))
223 return Insts::mult(args.at(0));
224 if (type == INST_FURI->resolve("mod"))
225 return Insts::mod(args.at(0));
226 if (type == INST_FURI->resolve("eq"))
227 return Insts::eq(args.at(0));
228 if (type == INST_FURI->resolve("neq"))
229 return Insts::neq(args.at(0));
230 if (type == INST_FURI->resolve("gte"))
231 return Insts::gte(args.at(0));
232 if (type == INST_FURI->resolve("gt"))
233 return Insts::gt(args.at(0));
234 if (type == INST_FURI->resolve("lte"))
235 return Insts::lte(args.at(0));
236 if (type == INST_FURI->resolve("lt"))
237 return Insts::lt(args.at(0));
238 if (type == INST_FURI->resolve("to"))
239 return Insts::to(args.at(0));
240 if (type == INST_FURI->resolve("from"))
241 return Insts::to(args.at(0));
242 if (type == INST_FURI->resolve("pub"))
243 return Insts::pub(args.at(0), args.at(1));
244 if (type == INST_FURI->resolve("sub"))
245 return Insts::sub(args.at(0), args.at(1));
246 if (type == INST_FURI->resolve("print"))
247 return Insts::print(args.at(0));

250 return Insts::bswitch(args.at(0));
251 if (type == INST_FURI->resolve("explain"))
252 return Insts::explain();
253 throw fError("Unknown instruction: %s\n", type.toString().c_str());
