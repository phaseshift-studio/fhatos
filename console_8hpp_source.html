35 public:
36 explicit Console(const ID &id = ID("console")) : Thread(id) {}

38 void setup() override { Thread::setup(); }

40 void loop() override {
41 Thread::loop();
42 string line;
43 while (true) {
44 this->printPrompt();
45 line.clear();
46 std::getline(std::cin, line);
47 StringHelper::trim(line);
49 if (line.empty()) {
50 // do nothing
51 } else if (line[0] == ':') {
52 if (line == ":quit") {
53 this->stop();
54 return;
55 } else if (strstr(line.c_str(), ":log ")) {
56 try {
57 if (line.length() < 6) {
58 this->printResult(Obj::to_str(LOG_TYPES.toChars((LOG_TYPE) GLOBAL_OPTIONS->LOGGING)));
59 } else {
60 string level = line.substr(5);
61 GLOBAL_OPTIONS->LOGGING = LOG_TYPES.toEnum(level.c_str());
62 }
63 } catch (const fError &e) {
64 this->printException(e);
65 }
66 } else if (strstr(line.c_str(), ":router ")) {
67 if (line.length() < 9) {
68 this->printResult(Obj::to_str(GLOBAL_OPTIONS->router<Router>()->toString()));
69 } else {
70 string router = line.substr(8);
71 if (router == "LocalRouter")
72 GLOBAL_OPTIONS->ROUTING = LocalRouter::singleton();
73 else if (router == "MqttRouter")
74 GLOBAL_OPTIONS->ROUTING = MqttRouter::singleton();
75 else
76 this->printException(fError("Invalid logger (LocalRouter,MqttRouter): %s\n", router.c_str()));
77 }
78 }
79 } else {
80 try {
81 const Option<Obj_p> obj = Parser::singleton()->tryParseObj(line);
82 if (obj.value()->isBytecode())
83 this->printResults(Fluent(obj.value()));
84 else
85 this->printResult(obj.value());
86 } catch (const std::exception &e) {
87 this->printException(e);
88 }
89 }
90 }
91 }
92 void stop() override { Thread::stop(); }
94 void printException(const std::exception &ex) const {
95 GLOBAL_OPTIONS->printer()->printf("!r[ERROR]!! %s", ex.what());
96 }
97 void printPrompt() const { GLOBAL_OPTIONS->printer()->print("!mfhatos!!> "); }
98 void printResults(const Fluent &fluent) const {
99 fluent.forEach<Obj>([this](const Obj_p &obj) {
100 if (obj->isLst()) {
101 for (const auto &o: *obj->lst_value()) {
102 this->printResult(o);
103 }
104 } else
105 this->printResult(obj);
106 });
107 }
108 void printResult(const Obj_p &obj) const {
109 GLOBAL_OPTIONS->printer()->printf("!g==>!!%s\n", obj->toString().c_str());
110 }
111 };
