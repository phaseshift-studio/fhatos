124 return (new string(string("Unknown error code: ") + std::to_string(rc)))->c_str();

166 template<typename OBJ = Obj>
167 static ptr<OBJ> read(const ID &target, const ID &source = FOS_DEFAULT_SOURCE_ID) {
168 Router *router = GLOBAL_OPTIONS->router<Router>();
169 auto *thing = new std::atomic<OBJ *>(nullptr);
170 auto *done = new std::atomic_bool(false);
171 router->subscribe(
172 Subscription{.source = source, .pattern = target, .onRecv = [thing, done](const Message_p &message) {
173 thing->store(new OBJ(*message->payload));
174 done->store(true);
175 }});
176 const time_t startTimestamp = time(nullptr);
177 while (!done->load()) {
178 if ((time(nullptr) - startTimestamp) > ((uint8_t) router->_level) + 1) {
179 LOG(ERROR, "Target undefined !y%s!!\n", target.toString().c_str());
180 break;
181 }
182 }
183 router->unsubscribe(source, target);
184 if (nullptr == thing->load()) {
185 delete thing;
186 delete done;
187 return Obj::to_noobj();
188 } else {
189 ptr<OBJ> ret = ptr<OBJ>(thing->load());
190 delete thing;
191 delete done;
192 return ret;
193 }
194 }
