29 static string hexStr(const unsigned char *data, const int len) {
30 string s(len * 2, ' ');
31 for (uint8_t i = 0; i < len; ++i) {
32 s[2 * i] = hexmap[(data[i] & 0xF0) >> 4];
33 s[2 * i + 1] = hexmap[data[i] & 0x0F];
34 }
35 return s;
36 }
