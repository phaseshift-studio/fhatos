29 static string hexStr(const unsigned char *data, const int len) {
30 string s(len * 2, ' ');
31 for (uint8_t i = 0; i < len; ++i) {
32 s[2 * i] = hexmap[(data[i] & 0xF0) >> 4];
33 s[2 * i + 1] = hexmap[data[i] & 0x0F];
34 }
35 return s;

51 static bool lookAhead(const string &token, std::stringstream *ss) {
52 for (int i = 0; i < token.size(); i++) {
53 if (token[i] != ss->peek()) {
54 for (int j = 0; j <= i; j++) {
55 ss += token[j];
56 }
57 return false;
58 } else {
59 ss->get();
60 }
61 }
62 return true;

64 static int split(const char *text, const char *deliminator, char **&result, const uint8_t offset = 0) {
65 char *copy;
66 const char *freeable_copy = copy = strdup(text);
67 char *token;
68 int i = offset;
69
70 while ((token = strsep(&copy, deliminator)) != nullptr) {
71 if (strlen(token) > 0) {
72 result[i] = strdup(token);
73 i++;
74 }
75 }
76 size_t dl = strlen(deliminator);
77 char *substr = new char[dl];
78 strncpy(substr, text + (strlen(text) - dl), dl);
79 substr[dl] = '\0';
80 if (strcmp(substr, deliminator) == 0) {
81 result[i] = strdup("");
82 i++;
83 }
84 // delete substr;
85 delete token;
86 delete freeable_copy;
87 return i;
88 }

90 static bool match(const char *id_cstr, const char *pattern_cstr) {
91 if (strstr(pattern_cstr, "#") == nullptr && strstr(pattern_cstr, "+") == nullptr)
92 return strcmp(id_cstr, pattern_cstr) == 0;
93 if (strlen(id_cstr) == 0 && strcmp(pattern_cstr, "#") == 0)
94 return true;
95 char **idParts = new char *[FOS_MAX_FURI_SEGMENTS];
96 char **patternParts = new char *[FOS_MAX_FURI_SEGMENTS];
97 int idLength = split(id_cstr, "/", idParts);
98 if (id_cstr[strlen(id_cstr) - 1] == '/')
99 idLength++;
100 const int patternLength = split(pattern_cstr, "/", patternParts);
101 // LOG(DEBUG, "Matching: %s <=> %s\n", id, pattern);
102 const bool result = [idParts, patternParts, idLength, patternLength]() {
103 for (int i = 0; i < idLength; i++) {
104 if (i >= patternLength)
105 return false;
106 // LOG(DEBUG, "\t%s <=%i=> %s\n", idParts[i], i, patternParts[i]);
107 if (strcmp(patternParts[i], "#") == 0)
108 return true;
109 if ((strcmp(patternParts[i], "+") != 0) && (strcmp(patternParts[i], idParts[i]) != 0))
110 return false;
111 }
112 return patternLength == idLength;
113 }();
114 for (uint8_t i = 0; i < idLength; i++) {
115 delete[] idParts[i];
116 }
117 delete[] idParts;
118 for (uint8_t i = 0; i < patternLength; i++) {
119 delete[] patternParts[i];
120 }
121 delete[] patternParts;
122 return result;
123 }
