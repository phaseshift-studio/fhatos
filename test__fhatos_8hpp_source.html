213template<typename OBJ = Obj>
214static void FOS_CHECK_RESULTS(const List<OBJ> &expected, const Fluent &fluent,
215 const Map<Uri, Obj, Obj::obj_comp> &expectedReferences = {},
216 const bool clearRouter = true) {
217 const ptr<List<ptr<OBJ>>> result = FOS_TEST_RESULT<OBJ>(fluent);
218 TEST_ASSERT_EQUAL_INT_MESSAGE(expected.size(), result->size(), "Expected vs. actual result size");
219 for (const OBJ &obj: expected) {
220 auto x = std::find_if(result->begin(), result->end(), [obj](const ptr<OBJ> element) {
221 if (obj.isReal()) {
222 return obj.real_value() + 0.01f > element->real_value() && obj.real_value() - 0.01f < element->real_value();
223 } else
224 return obj == *element;
225 });
226 if (result->end() == x) {
227 TEST_FAIL_MESSAGE(("Unable to find " + obj.toString()).c_str());
228 }
229 }
230 if (!expectedReferences.empty()) {
231 if (GLOBAL_OPTIONS->router<Router>()->retainSize() != -1) {
232 TEST_ASSERT_EQUAL_INT_MESSAGE(expectedReferences.size(), GLOBAL_OPTIONS->router<Router>()->retainSize(),
233 "Expected vs. actual router retain message size");
234 for (const auto &[key, value]: expectedReferences) {
235 const Obj temp = value;
236 GLOBAL_OPTIONS->router<Router>()->subscribe(Subscription{
237 .mailbox = nullptr,
238 .source = ID(FOS_DEFAULT_SOURCE_ID),
239 .pattern = key.uri_value(),
240 .onRecv = [temp](const ptr<Message> &message) { TEST_ASSERT_TRUE(temp == *message->payload); }});
241 }
242 }
243 }
244 if (clearRouter)
245 GLOBAL_OPTIONS->router<Router>()->clear();
246}
