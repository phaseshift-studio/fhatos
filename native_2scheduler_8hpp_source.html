46 bool spawn(Process *process) override {
47 bool success = *RW_PROCESS_MUTEX.write<bool>([this, process]() {
48 // TODO: have constructed processes NOT running or check is process ID already in scheduler
49 process->setup();
50 if (!process->running()) {
51 LOG_TASK(ERROR, this, "!RUnable to spawn running %s: %s!!\n", P_TYPE_STR(process->type),
52 process->id()->toString().c_str());
53 return share(false);
54 }
57 bool success = false;
58 this->subscribe(*process->id(), [this, process](const Message_p &message) {
59 if (message->payload->isNoObj()) {
60 process->stop();
61 this->unsubscribe(*process->id());
62 }
63 });
64 switch (process->type) {
65 case THREAD: {
66 this->THREADS->push_back(static_cast<Thread *>(process));
67 dynamic_cast<Thread *>(process)->xthread = new std::thread(&Scheduler::THREAD_FUNCTION, process);
68 success = true;
69 break;
70 }
71 case FIBER: {
72 success = this->FIBERS->push_back(dynamic_cast<Fiber *>(process));
73 LOG(INFO, "Fiber bundle count: %i\n", this->FIBERS->size());
74 if (!FIBER_THREAD_HANDLE) {
75 FIBER_THREAD_HANDLE = new std::thread(&Scheduler::FIBER_FUNCTION, nullptr);
76 if (FIBER_THREAD_HANDLE)
77 success = true;
78 }
79 dynamic_cast<Fiber *>(process)->xthread = FIBER_THREAD_HANDLE;
80 break;
81 }
82 case COROUTINE: {
83 success = this->COROUTINES->push_back(dynamic_cast<Coroutine *>(process));
84 break;
85 }
86 case KERNEL: {
87 success = this->KERNELS->push_back(dynamic_cast<KernelProcess *>(process));
88 break;
89 }
90 default: {
91 LOG_TASK(ERROR, this, "!b%s!! has an unknown process type: !r%i!!\n", process->id()->toString().c_str(),
92 process->type);
93 return share(false);
94 }
95 }
96 LOG_TASK(success ? INFO : ERROR, this, "!b%s!! !y%s!! spawned\n", process->id()->toString().c_str(),
97 P_TYPE_STR(process->type));
98 /*LOG(NONE,
99 "\t!yFree memory\n"
100 "\t !b[inst:" FOS_BYTES_MB_STR "][heap: " FOS_BYTES_MB_STR "][psram: " FOS_BYTES_MB_STR "][flash: "
101 FOS_BYTES_MB_STR "]\n",
102 FOS_BYTES_MB(ESP.getFreeSketchSpace()),
103 FOS_BYTES_MB(ESP.getFreeHeap()),
104 FOS_BYTES_MB(ESP.getFreePsram()),
105 FOS_BYTES_MB(ESP.getFlashChipSize()));*/
106 /* if (success) {
107 this->publish(
108 *this->id(),
109 Obj::to_rec(
110 {{u("thread"), THREADS->size()}, {u("fiber"), FIBERS->size()}, {u("coroutine"),
111 COROUTINES->size()}}), RETAIN_MESSAGE);
112 }*/
113 return share(success);
114 });
115 if (!success)
116 this->unsubscribe(*process->id());
117 return success;
118 }

120 std::thread *FIBER_THREAD_HANDLE = nullptr;
124 static void FIBER_FUNCTION(void *voidptr) {
125 auto *fibers = Scheduler::singleton()->FIBERS;
126 int counter = 0;
127 while (!fibers->empty()) {
128 Option<Fiber *> fiber = fibers->get(counter);
129 if (fiber.has_value()) {
130 if (!fiber.value()->running())
131 Scheduler::singleton()->destroy(*fiber.value()->id());
132 else
133 fiber.value()->loop();
134 counter = (counter + 1) % fibers->size();
135 } else {
136 counter = 0;
137 }
138 }
139 }
140
144 static void THREAD_FUNCTION(void *vptr_thread) {
145 auto *thread = static_cast<Thread *>(vptr_thread);
146 while (thread->running()) {
147 thread->loop();
148 }
149 Scheduler::singleton()->_destroy(*thread->id());
150 }
