29 template<typename ROUTER = FOS_DEFAULT_ROUTER >
30 struct PingRoutine final : public Coroutine, public Publisher<ROUTER> {
31 string ip;
32 uint16_t counter = 0;
33 uint16_t success = 0;
34 float totalTime = 0.0f;
35
36 explicit PingRoutine(const ID &id) : Coroutine(id), Publisher<ROUTER>(this) {
37 this->ip = fWIFI::resolve(this->id().path().c_str());
38 }
39
40 void setup() override {
41 Coroutine::setup();
42 this->onQuery(this->id().query("?loop"), [this]() { this->loop(); });
43 }
44
45 void stop() override {
46 this->unsubscribe(this->id().query("?loop"));
47 }
48
49 void loop() override {
50 Coroutine::loop();
51 ++this->counter;
52 char message[100];
53 if (Ping.ping(this->id().path().c_str(), 1)) {
54 ++this->success;
55 this->totalTime += Ping.averageTime();
56 sprintf(message, "64 bytes from %s [%s]: icmp_seq=%i time=%.3f ms",
57 this->id().path().c_str(), this->ip.c_str(), this->counter,
58 this->averageTime());
59 } else {
60 sprintf(
61 message,
62 "Request timeout for %s [unknown] icmp_seq %i failure_rate=%.2f%%",
63 this->id().path().c_str(), this->counter, this->failureRate());
64 }
65 this->publish(this->id(), new BinaryObj(message), TRANSIENT_MESSAGE);
66 }
67
68 float failureRate() const {
69 return (static_cast<float>(counter - success) / static_cast<float>(counter)) * 100.0f;
70 }
71
72 float averageTime() const {
73 return this->totalTime / (float) this->counter;
74 }
75 };

117 List<Process *> *results = Scheduler::singleton()->find(this->id().extend("+"));
