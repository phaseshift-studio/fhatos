121 fWIFI *setStation() {
122 LOG(INFO, "!b[WIFI Station Configuration]!!\n");
123 WiFiClass::mode(WIFI_STA);
124 WiFi.setAutoReconnect(true);
125 // WiFi.persistent(true);
126 const char *delim = ":";
127 char ssidsTemp[strlen(this->ssids) + 1];
128 sprintf(ssidsTemp, "%s", this->ssids);
129 char *ssid = strtok(ssidsTemp, delim);
130 int i = 0;
131 char *ssids_parsed[10];
132 LOG(NONE, "\tWiFi SSIDs:\n");
133 while (ssid != nullptr) {
134 LOG(NONE, "\t\t%s\n", ssid);
135 ssids_parsed[i++] = ssid;
136 ssid = strtok(nullptr, delim);
137 }
138 i = 0;
139 char passwordsTemp[50];
140 sprintf(passwordsTemp, "%s", this->passwords);
141 char *passwords_parsed[10];
142 char *password = strtok(passwordsTemp, delim);
143 while (password != nullptr) {
144 passwords_parsed[i++] = password;
145 password = strtok(nullptr, delim);
146 }
147 WIFI_MULTI_CLIENT multi;
148 for (int j = 0; j < i; j++) {
149 multi.addAP(ssids_parsed[j], passwords_parsed[j]);
150 }
151 uint8_t attempts = 0;
152 while (attempts < 10) {
153 attempts++;
154 if (multi.run() == WL_CONNECTED) {
155 this->_id =
156 ptr<ID>(new ID(fWIFI::idFromIP("kernel", "wifi").toString().c_str()));
157 WiFiClass::hostname(this->id()->user().value().c_str());
158 const bool mdnsStatus = MDNS.begin(this->id()->user().value().c_str());
159 LOG(NONE,
160 "\tID : %s\n"
161 "\tStatus : %s\n"
162 "\tSSID : %s\n"
163 "\tMAC address : %s\n"
164 "\tIP address : %s\n"
165 "\tHostname : %s\n"
166 "\tmDNS name : %s\n"
167 "\tGateway address: %s\n"
168 "\tSubnet mask : %s\n"
169 "\tDNS address : %s\n"
170 "\tChannel : %i\n",
171 this->id()->toString().c_str(),
172 WiFi.isConnected() ? "CONNECTED" : "DISCONNECTED",
173 WiFi.SSID().c_str(), WiFi.macAddress().c_str(),
174 WiFi.localIP().toString().c_str(), WiFi.getHostname(),
175 mdnsStatus ? (this->id()->user().value() + ".local").c_str()
176 : "<error>",
177 WiFi.gatewayIP().toString().c_str(),
178 WiFi.subnetMask().toString().c_str(),
179 WiFi.dnsIP().toString().c_str(), WiFi.channel());
180 if (!mdnsStatus) {
181 LOG_TASK(ERROR, this, "Unable to create mDNS hostname %s\n",
182 this->id()->user()->c_str());
183 }
184 LOG(INFO, "\tConnection attempts: %i\n", attempts);
185 attempts = 100;
186 }
187 }
188 if (attempts != 100) {
189 LOG_TASK(ERROR, this, "Unable to connect to WIFI after %i attempts\n",
190 attempts);
191 }
192 return this;
193 }
