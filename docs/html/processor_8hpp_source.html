42 List<const Monad *> split(const ptr<Bytecode> bcode) const {
43 if (this->_inst->isNoInst() || this->_obj->isNoObj()) {
44 return List<const Monad *>{};
45 } else {
46 const Obj *nextObj = this->_inst->apply(this->_obj);

77 template<typename E>
78 class Processor {
79 protected:
80 const ptr<Bytecode> bcode;
81 List<const Monad *> *running = new List<const Monad *>();
82 List<const E *> *halted = new List<const E *>();
83 Pair<ManyToOneInst *, List<const Obj *> *> *barrier;
84
85 public:
86 explicit Processor(const ptr<Bytecode> bcode) : bcode(bcode),
87 barrier(new Pair<ManyToOneInst *, List<const Obj *> *>(
88 nullptr, nullptr)) {
89 const Inst *startInst = this->bcode->startInst();
90 assert(startInst->opcode() == "start");
91 for (const Obj *startObj: startInst->args()) {
92 const Monad *monad = new Monad(startObj, startInst);
93 this->running->push_back(monad);
94 LOG(DEBUG, FOS_TAB_2 "!mStarting!! monad: %s\n", monad->toString().c_str());
95 }
96 }
97
98 const E *next(const int steps = -1) {
99 while (true) {
100 if (this->halted->empty()) {
101 if (this->running->empty()) {
102 return nullptr;
103 } else {
104 this->execute(steps);
105 }
106 } else {
107 const E *end = this->halted->back();
108 this->halted->pop_back();
109 return end;
110 }
111 }
112 }
113
114 int execute(const int steps = -1) {
115 int counter = 0;
116
117 while ((!this->running->empty() || this->barrier->first) && (counter++ < steps || steps == -1)) {
118 if (this->running->empty() && this->barrier->first) {
119 const Obj *objA = new Objs(this->barrier->second);
120 LOG(DEBUG, "Processing barrier: %s\n", objA->toString().c_str());
121 const Obj *objB = this->barrier->first->apply(objA);
122 LOG(DEBUG, "Barrier reduction: %s\n", objB->toString().c_str());
123 this->running->push_back(new Monad(objB, bcode->nextInst(this->barrier->first)));
124 this->barrier->second->clear();
125 this->barrier->first = nullptr;
126 delete this->barrier->second;
127 } else {
128 const Monad *parent = this->running->back();
129 this->running->pop_back();
130 if (parent->dead()) {
131 LOG(DEBUG, FOS_TAB_4 "!rKilling!! monad: %s\n", parent->toString().c_str());
132 } else if (parent->halted()) {
133 LOG(DEBUG, FOS_TAB_2 "!gHalting!! monad: %s\n", parent->toString().c_str());
134 this->halted->push_back((const E *) parent->obj());
135 } else {
136 if (parent->inst()->itype() == IType::MANY_TO_ONE) {
138 if (!this->barrier->first) {
139 LOG(DEBUG, "Creating barrier: %s\n", parent->inst()->toString().c_str());
140 this->barrier->first = (ManyToOneInst *) parent->inst();
141 this->barrier->second = new List<const Obj *>();
142 }
143 LOG(DEBUG, "Adding to barrier: %s => %s\n", parent->toString().c_str(),
144 parent->inst()->toString().c_str());
145 this->barrier->second->push_back(parent->obj());
146 } else {
147 for (const Monad *child: parent->split(this->bcode)) {
148 LOG(DEBUG, FOS_TAB_4 "!ySplitting!! monad : %s => %s\n", parent->toString().c_str(),
149 child->toString().c_str());
150 this->running->push_back(child);
151 }
152 }
153 }
154 delete parent;
155 }
156 }
157
158 LOG(DEBUG, FOS_TAB_2 "Exiting current run with [!yrunning!!:%i] [!ghalted!!:%i]\n",
159 this->running->size(),
160 this->halted->size());
161 return this->halted->size();
162 }

164 void forEach(const Consumer<const E *> &consumer, const int steps = -1) {
165 while (true) {
166 const E *end = this->next();
167 if (end) {
168 consumer(end);
169 } else {
170 break;
171 }
172 }
173 }
174 };
