134 template<typename ROUTER = FOS_DEFAULT_ROUTER >
135 class DereferenceInst final : public OneToOneInst {
136 public:
137 explicit DereferenceInst(const URI_OR_BYTECODE &uri, const URI_OR_BYTECODE &source) : OneToOneInst(
138 "dref", {uri, source}, [this](const Obj *obj) -> const Obj *{
139 std::atomic<Obj *> *thing = new std::atomic<Obj *>(nullptr);
140 std::atomic<bool> *done = new std::atomic<bool>(false);
141 const fURI uri = this->arg(0)->apply(obj)->template as<Uri>()->value();
142 const fURI source = this->arg(1)->apply(obj)->template as<Uri>()->value();
143 ROUTER::singleton()->subscribe(
144 Subscription{
145 .source = source,
146 .pattern = uri,
147 .onRecv = [thing,done](const Message &message) {
148 thing->store(message.payload->toObj());
149 done->store(true);
150 }
151 });
152 while (!done->load()) {
153 // wait until done
154 }
155 const Obj *ret = thing->load();
156 delete thing;
157 delete done;
158 ROUTER::singleton()->unsubscribe(source, uri);
159 return ret;
160 }) {
161 }
162 };
