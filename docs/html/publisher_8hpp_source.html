64 this->subscribe(queryPattern, [this,queryPattern,consumer](const Message &message) {
65 if (!message.source.equals(this->ided->id()) && message.isQuery(queryPattern.query().c_str()))

71 Publisher<ROUTER> *onQuery(const Pattern &queryPattern,
72 const Map<string, BiConsumer<SourceID, TargetID> > &mapping,
73 const BiConsumer<SourceID, TargetID> &thenDo = nullptr) {
74 this->subscribe(queryPattern, [this,mapping,thenDo](const Message &message) {
75 if (message.target.hasQuery() && !message.source.equals(this->ided->id())) {
76 if (const string query = message.target.query(); mapping.count(query)) {
77 mapping.at(query)(message.source, message.target);
78 } else if (mapping.count("default")) {
79 mapping.at("default")(message.source, message.target);
80 }
81 if (thenDo)
82 thenDo(message.source, message.target);
83 }
84 });
85 return this;
86 }

88 Publisher<ROUTER> *onQuery(const Pattern &queryPattern,
89 const Map<string, Obj> &mapping) {
90 Map<string, BiConsumer<SourceID, TargetID> > map;
91 for (const auto pair: mapping) {
92 map.insert({
93 {
94 pair.first, [this,pair](const SourceID source, const TargetID target) {
95 this->publish(target, BinaryObj<>::fromObj(&pair.second),RETAIN_MESSAGE);
96 }
97 }
98 });
99 }
100 return this->onQuery(queryPattern, map);
101 }
