33 public:
34 explicit fError(const char *format, ...) {
35 va_list arg;
36 va_start(arg, format);
37 char temp[64];
38 _message = temp;
39 size_t len = vsnprintf(temp, sizeof(temp), format, arg);
40 va_end(arg);
41 if (len > sizeof(temp) - 1) {
42 _message = new (std::nothrow) char[len + 1];
43 if (!_message) {
44 return;
45 }
46 va_start(arg, format);
47 vsnprintf(_message, len + 1, format, arg);
48 va_end(arg);
49 }
50 if (_message != temp) {
51 delete[] _message;
52 }
53 };
