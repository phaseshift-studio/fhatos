93 const void saveType(const ID &typeId, const Obj_p &obj, const bool writeThrough = true) const {
94 CACHE_MUTEX->write<void>([this, typeId, obj, writeThrough] {
95 CACHE->erase(typeId);
96 if (!obj->isNoObj()) {
97 CACHE->insert({typeId, PtrHelper::clone<Obj>(obj)});
98 if (writeThrough)
99 Router::write(typeId, obj);
100 if (OType::INST == OTypes.toEnum(typeId.path(0, 1).c_str())) {
101 const Inst_p inst = Insts::to_inst(typeId, *obj->lst_value());
102 LOG_TASK(INFO, this, "!b%s!g[!!%s!g]!m:!b%s !ytype!! defined\n", typeId.toString().c_str(),
103 obj->lst_value()->front()->toString().c_str(), ITypeSignatures.toChars(inst->itype()));
104 } else {
105 LOG_TASK(INFO, this, "!b%s!g[!!%s!g] !ytype!! defined\n", typeId.toString().c_str(),
106 obj->toString().c_str());
107 }
108 }
109 return share(nullptr);
110 });

112 const Option<Obj_p> loadType(const ID &typeId, const bool readThrough = true) const {
113 return CACHE_MUTEX->read<Option<Obj_p>>([this, typeId, readThrough] {
114 try {
115 if (CACHE->count(typeId) && !CACHE->at(typeId)->isNoObj())
116 return Option<Obj_p>(CACHE->at(typeId));
117 if (readThrough) {
118 const Type_p type = Router::read<Obj>(typeId);
119 return type->isNoObj() ? Option<Obj_p>() : Option<Obj_p>(type);
120 }
121 } catch (const fError &) {
122 }
123 return Option<Obj_p>();
124 });

127 bool checkType(const Obj &obj, const OType otype, const ID &typeId, const bool doThrow = true) const
128 noexcept(false) {
129 const OType typeOType = OTypes.toEnum(typeId.path(0, 1).c_str());
130 if (otype == OType::INST || otype == OType::BCODE || typeOType == OType::INST || typeOType == OType::BCODE)
131 return true;
132 if (otype != typeOType) {
133 if (doThrow)
134 throw fError("!g[!b%s!g]!! %s is not a !b%s!!\n", Types::singleton()->id()->toString().c_str(),
135 obj.toString().c_str(), typeId.toString().c_str());
136 return false;
137 }
138 if (typeId.pathLength() == 2 && typeId.lastSegment().empty()) {
139 return true;
140 }
141 const Option<Type_p> type = loadType(typeId);
142 if (type.has_value()) {
143 if (obj.match(type.value(), false)) {
144 return true;
145 }
146 if (doThrow)
147 throw fError("!g[!b%s!g]!! %s is not a !b%s!g[!!%s!g]!!\n", Types::singleton()->id()->toString().c_str(),
148 obj.toString().c_str(), typeId.toString().c_str(), type.value()->toString().c_str());
149 return false;
150 }
151 if (doThrow)
152 throw fError("!g[!b%s!g] !b%s!! is an undefined !ytype!!\n", Types::singleton()->id()->toString().c_str(),
153 typeId.toString().c_str());
154 return false;
155 }
