26 using PriorPost = Pair<List<ID>, List<ID>>;
27 using Rewrite = Trip<ID, Function<BCode_p, BCode_p>, PriorPost>;
28 struct Rewriter {
29 List<Rewrite> _rewrites;
30 explicit Rewriter(const List<Rewrite> &rewrites) : _rewrites(rewrites) {}
31 BCode_p apply(const BCode_p &bcode) {
32 BCode_p running = bcode;
33 for (const Rewrite &rw: this->_rewrites) {
34 LOG(DEBUG, "Applying rewrite %s\n", std::get<0>(rw).toString().c_str());
35 running = std::get<1>(rw)(running);
36 }
37 return running;
38 }
39 static void LOG_REWRITE(const ID &rewriteID, const BCode_p &original, const BCode_p &rewrite) {
40 LOG(DEBUG, "!g[!b%s!g]!! !yrewrote!! %s !r=to=>!! %s\n", rewriteID.toString().c_str(),
41 original->toString().c_str(), rewrite->toString().c_str());
42 }
43 static Rewrite explain() {
44 return Rewrite({ID("/lang/rewrite/explain"),
45 [](const BCode_p &bcode) {
46 if (bcode->bcode_value().back()->id()->equals(ID("/inst/explain"))) {
47 string ex;
48 for (const Inst_p &inst: bcode->bcode_value()) {
49 ex += inst->toString() + "\t" + ITypeSignatures.toChars(inst->itype()) + "\n";
50 }
51 bcode->bcode_value().back()->inst_seed()->add_obj(Obj::to_str(ex));
52 }
53 return bcode;
54 },
55 {{}, {}}});
56 }
57 static Rewrite by() {
58 return Rewrite({ID("/lang/rewrite/by"),
59 [](const BCode_p &bcode) {
60 Inst_p prev = Obj::to_noobj();
61 bool found = false;
62 List<Inst_p> newInsts;
63 for (const Inst_p &inst: bcode->bcode_value()) {
64 if (inst->id()->equals(ID("/inst/by")) && !prev->isNoObj()) {
65 found = true;
66 // rewrite args
67 bool done = false;
68 List<Obj_p> newArgs;
69 for (const Obj_p &arg: prev->inst_args()) {
70 if (!done && arg->isNoObj()) {
71 newArgs.push_back(inst->inst_arg(0));
72 done = true;
73 } else {
74 newArgs.push_back(arg);
75 }
76 }
77 if (!done)
78 throw fError("Previous inst could not be by()-modulated: %s !r<=/=!! %s\n",
79 prev->toString().c_str(), inst->toString().c_str());
80 // rewrite inst
81 newInsts.pop_back();
82 newInsts.push_back(Insts::to_inst(*prev->id(), newArgs));
83 } else {
84 newInsts.push_back(inst);
85 }
86 prev = newInsts.back();
87 }
88 if (found) {
89 const BCode_p rewrite = Obj::to_bcode(newInsts);
90 LOG_REWRITE(ID("/lang/rewrite/by"), bcode, rewrite);
91 return rewrite;
92 }
93 return bcode;
94 },
95 {{}, {}}});
96 }
97 };
